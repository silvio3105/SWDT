190 void setTimeout(uint32_t timeout)
191 {
192 // Set prescaler to max
193 setPrescaler(Prescaler_t::Div256);
194
195 // Set required reload value
196 setReloadValue(timeout / (1000 / (freq / 256)));
197 }
198
199 private:
200 // CONSTANTS
201 static constexpr uint16_t reloadKey = 0xAAAA;
202 static constexpr uint16_t accessKey = 0x5555;
203 static constexpr uint16_t startKey = 0xCCCC;
204 static constexpr uint16_t maxReloadValue = 4095;
206 // ENUMS
211 enum class Prescaler_t : uint8_t {
212 Div4 = 0b000,
213 Div8 = 0b001,
214 Div16 = 0b010,
215 Div32 = 0b011,
216 Div64 = 0b100,
217 Div128 = 0b101,
218 Div256 = 0b110
219 };

227 inline void enableAccess(void)
228 {
229 // Write access value to KR register to unlock register protection
230 IWDG->KR = accessKey;
231 }

239 void setReloadValue(uint32_t value)
240 {
241 // Prevent reload value to go over its maximum
242 SML::limit<uint32_t>(value, 1, maxReloadValue);
243
244 // Wait if reload value update is ongoing
245 // while (IWDG->SR & IWDG_SR_RVU);
246
247 // Enable write access
248 enableAccess();
249
250 // Write new reload value
251 IWDG->RLR = value;
252 }
253
259 void setPrescaler(const Prescaler_t prescaler)
260 {
261 // Wait if prescaler update is ongoing
262 // while (IWDG->SR & IWDG_SR_PVU);
263
264 // Enable write access
265 enableAccess();
266
267 // Set new prescaler
268 IWDG->PR = (uint8_t)prescaler;
269 }
